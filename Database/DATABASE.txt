#1 RIVER_MANAGEMENT
CREATE TABLE RIVER_MANAGEMENT
(MANAGEMENT_ID NUMBER(3),
NAME VARCHAR2(15),
SOURCE VARCHAR2(15),
X NUMBER(10,4),
Y NUMBER(10,4),
PRIMARY KEY (MANAGEMENT_ID));


#2 ENTRANCE_TO_GREENSPACE
CREATE TABLE ENTRANCE_TO_GREENSPACE
(ID NUMBER(3),
X NUMBER(10,4),
Y NUMBER(10,4),
PRIMARY KEY (ID));


#3 DATA_ZONE
CREATE TABLE DATA_ZONE
(DATA_ZONE NUMBER(3),
NAME VARCHAR2(15),
AREA NUMBER(4,4),
POPULATION NUMBER(6),
SIMD_RANK_2020 NUMBER(4),
NO_OF_ACCESSIBLE_GS NUMBER(3),
TOT_AREA_OF_ACCESSIBLE_GS NUMBER(4,4),
ACCESSIBLE_GS_PER_PERSON NUMBER(4,4),
PRIMARY KEY (DATA_ZONE));


#4 GREENSPACE
CREATE TABLE GREENSPACE
(GS_ID NUMBER(3),
FUNCTION VARCHAR2(15),
AREA NUMBER(4,4),
DATA_ZONE NUMBER(3),
PRIMARY KEY (GS_ID),
FOREIGN KEY (DATA_ZONE));


# Relation Table
CREATE TABLE DATA_ZONE_GREENSPACE
(ID NUMBER(3),
DATA_ZONE NUMBER(3),
GS_ID NUMBER(3),
PRIMARY KEY (ID));
